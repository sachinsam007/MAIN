<script>
  const fmt = (n) => '₹' + Math.round(n).toLocaleString('en-IN');

  function calculate() {
    const onRoad = parseFloat(document.getElementById('onRoad').value) || 0;
    const downPayment = parseFloat(document.getElementById('downPayment').value) || 0;
    const pf = parseFloat(document.getElementById('pf').value) || 0;
    const dc = parseFloat(document.getElementById('dc').value) || 0;
    const sc = parseFloat(document.getElementById('serviceCharge').value) || 0;
    const roi = parseFloat(document.getElementById('roi').value) || 0;    const tenure = parseInt(document.getElementById('tenure').value) || 0;

    // Disbursement = On Road - Down Payment
    const disbursement = onRoad - downPayment;
    document.getElementById('disbursement').value = disbursement > 0 ? Math.round(disbursement) : '';

    // Loan Amount = Disbursement + PF + DC + SC
    const pfAmt = disbursement * (pf / 100);
    const dcAmt = disbursement * (dc / 100);
    const loanAmount = disbursement + pfAmt + dcAmt + sc;
    document.getElementById('loanAmount').value = disbursement > 0 ? Math.round(loanAmount) : '';

    // Update breakdown
    document.getElementById('b-disb').textContent = disbursement > 0 ? fmt(disbursement) : '—';
    document.getElementById('b-pf').textContent = disbursement > 0 ? fmt(pfAmt) : '—';
    document.getElementById('b-dc').textContent = disbursement > 0 ? fmt(dcAmt) : '—';
    document.getElementById('b-sc').textContent = sc > 0 ? fmt(sc) : '—';

    // EMI calculation
    const resultCard = document.getElementById('resultCard');
    const emiDisplay = document.getElementById('emiDisplay');
    const extraInfo = document.getElementById('extraInfo');

    if (loanAmount > 0 && roi > 0 && tenure > 0) {
      // Flat rate EMI = (Principal + Total Interest) / Tenure
      // Total Interest = Principal * flat rate% * (tenure in years)
      const tenureYears = tenure / 12;
      const totalInterest = loanAmount * (roi / 100) * tenureYears;
      const totalPayable = loanAmount + totalInterest;
      const emi = totalPayable / tenure;

      emiDisplay.innerHTML = `
        <div class="emi-amount animate-pop">
          <span class="rupee">₹</span>${Math.round(emi).toLocaleString('en-IN')}
        </div>`;
      
      document.getElementById('totalPayable').textContent = fmt(totalPayable);
      document.getElementById('totalInterest').textContent = fmt(totalInterest);
      document.getElementById('tenureDisplay').textContent = tenure + ' mo';
      
      extraInfo.style.display = 'block';
      resultCard.classList.add('active');
    } else {
      emiDisplay.innerHTML = `<div class="emi-placeholder">${
        disbursement <= 0 ? 'Enter On Road & Down Payment' :
        !roi ? 'Select Rate of Interest' :
        !tenure ? 'Select Tenure' : 'Enter loan details above'
      }</div>`;
      extraInfo.style.display = 'none';
      resultCard.classList.remove('active');
    }
  }

  function resetAll() {
    ['onRoad','downPayment'].forEach(id => document.getElementById(id).value = '');
    document.getElementById('roi').value = '';
    document.getElementById('pf').value = '3.54';
    document.getElementById('dc').value = '4';
    document.getElementById('serviceCharge').value = '800';
    document.getElementById('tenure').value = '';
    document.getElementById('disbursement').value = '';
    document.getElementById('loanAmount').value = '';
    document.getElementById('emiDisplay').innerHTML = '<div class="emi-placeholder">Enter loan details above</div>';
    document.getElementById('extraInfo').style.display = 'none';
    document.getElementById('resultCard').classList.remove('active');
    ['b-disb','b-pf','b-dc','b-sc','totalPayable','totalInterest','tenureDisplay']
      .forEach(id => document.getElementById(id).textContent = '—');
  }
</script>
